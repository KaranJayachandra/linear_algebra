\begin{definition}
    A \textbf{Linear Combination} of variables $x_1, x_2, \ldots, x_n$ is of the form
    \begin{equation}
        \sum_{i=1}^N a_i x_i
    \end{equation}
    where $x_i \neq x_j$. The values $a_i \; \forall \; i \in (1, N)$ are the coeffecients of the combination.
\end{definition}
